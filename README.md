## Install

```bash
npm install --save livepeerjs-ima
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'livepeerjs-ima'
import 'livepeerjs-ima/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
